=== Code
```gdscript
extends Control

static var started : bool = false


func _ready() -> void:
	if started:
		%Player.position = Player.restart_position
	else:
		started = true
		Player.restart_position = %Player.position


func restart() -> void:
	if not is_inside_tree(): return
	get_tree().reload_current_scene()


func _on_player_failed() -> void:
	await get_tree().create_timer(2).timeout
	restart()


func _on_player_reached(area: Area2D) -> void:
	Player.restart_position = area.position

```
