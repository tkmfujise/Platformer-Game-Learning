== src / stages / gimmicks / falling_ice_block / falling_ice_block.gd

```gdscript
extends RigidBody2D


func _ready() -> void:
	GameEvent.falied.connect(_on_failed)
	freeze = true


func fall() -> void:
	freeze = false


func stop() -> void:
	freeze = true


func _on_detection_area_2d_body_entered(body: Node2D) -> void:
	fall.call_deferred()


func _on_failed() -> void:
	stop.call_deferred()

```
