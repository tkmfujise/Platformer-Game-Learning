```gdscript
extends Sprite2D

@export var on_fire : bool = true

func _ready() -> void:
	if on_fire: fire()
	else: wood()


func fire() -> void:
	%AnimationPlayer.play("fire")


func wood() -> void:
	%AnimationPlayer.play("wood")
```
