extends Sprite2D

@export var flicked: bool = false
@export var image : Texture2D : set = set_image


func set_image(image: Texture2D) -> void:
	%Image.texture = image


func flick() -> void:
	if not flicked: %AnimationPlayer.play("flick")


func pull() -> void:
	if flicked: %AnimationPlayer.play("pull")
