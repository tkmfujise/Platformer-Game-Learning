=== link:/scripts/src/stages/gimmicks/falling_ice_block/falling_ice_block.gd[falling_ice_block.gd]

```gdscript
extends RigidBody2D


func _ready() -> void:
	freeze = true
	%FailureCollisionShape2D.disabled = true


func _process(_delta: float) -> void:
	if not Player.playable: freeze = true


func fall() -> void:
	await get_tree().create_timer(0.5).timeout
	freeze = false
	%FailureCollisionShape2D.disabled = false


func _on_detection_area_2d_body_entered(_body: Node2D) -> void:
	fall.call_deferred()

```
